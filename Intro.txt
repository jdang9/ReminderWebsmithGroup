Reminder App @ Websmith Group

CREATE TABLE [dbo].[Users] (
    [UserID] int IDENTITY(1,1) NOT NULL,
    [UserName] nvarchar(max)  NOT NULL,
    [Password] nvarchar(max)  NOT NULL,
    [Email] nvarchar(max)  NOT NULL,
    [Admin] bit  NOT NULL,
    [Name] nvarchar(max)  NOT NULL
);
GO

CREATE TABLE [dbo].[MessageSlips] (
    [mId] int IDENTITY(1,1) NOT NULL,
    [UserID] nvarchar(max)  NOT NULL,
    [Property1] nvarchar(max)  NOT NULL,
    [Sender] nvarchar(max)  NOT NULL,
    [TimeDate] nvarchar(max)  NOT NULL,
    [PhoneNum] nvarchar(max)  NULL,
    [MessageTask] nvarchar(max)  NOT NULL,
    [Location] nvarchar(max)  NULL,
    [Other] nvarchar(max)  NULL,
    [User_UserID] int  NOT NULL
);
GO
