	[userID] int IDENTITY(1,1) NOT NULL, 
    [userName] nvarchar(30)  NOT NULL,
    [password] nvarchar(max)  NOT NULL,
    [email] nvarchar(max)  NULL,
    [admin] bit  NOT NULL,
    [firstName] nvarchar(max)  NOT NULL,
	[lastName] nvarchar(max)  NOT NULL,
	PRIMARY KEY ([userName])
